**3. Add reminder state and helper functions**

After the DOM refs and before `resetEventEntryForm`, add:

```javascript
let currentReminders = [30, 2880]; // default: 30 min + 2 days
const MAX_REMINDERS = 5;

function formatReminderMinutes(minutes) {
  if (minutes >= 1440 && minutes % 1440 === 0) return (minutes / 1440) + 'd';
  if (minutes >= 60 && minutes % 60 === 0) return (minutes / 60) + 'h';
  return minutes + ' min';
}

function renderReminderChips() {
  eventEntryReminderChips.innerHTML = '';
  currentReminders.forEach((minutes, index) => {
    const chip = document.createElement('span');
    chip.className = 'inline-flex items-center gap-1 rounded-full bg-indigo-500/25 border border-indigo-400/30 px-2.5 py-1 text-xs text-white/90';
    chip.innerHTML = `${formatReminderMinutes(minutes)} <button type="button" data-idx="${index}" class="ml-0.5 text-white/50 hover:text-white" aria-label="${I18N.reminderRemoveLabel}">&times;</button>`;
    chip.querySelector('button').addEventListener('click', () => {
      currentReminders.splice(index, 1);
      renderReminderChips();
      updateReminderAddState();
    });
    eventEntryReminderChips.appendChild(chip);
  });
}

function updateReminderAddState() {
  const atMax = currentReminders.length >= MAX_REMINDERS;
  eventEntryReminderAddBtn.disabled = atMax;
  eventEntryReminderAddBtn.classList.toggle('opacity-40', atMax);
  eventEntryReminderAddBtn.classList.toggle('cursor-not-allowed', atMax);
  eventEntryReminderCounter.classList.toggle('hidden', !atMax);
}

function setRemindersEnabled(enabled) {
  eventEntryRemindersContainer.classList.toggle('hidden', !enabled);
  eventEntryRemindersToggleLabel.textContent = enabled ? I18N.reminderToggleOn : I18N.reminderToggleOff;
}
```
